  <script>

    // 使用 GitHub Pages 路径（无需 raw.githubusercontent.com）

    const fileUrl = "https://nina999.github.io/template_file/";

    

    function downloadFile() {

      const link = document.createElement('a');

      link.href = fileUrl;

      link.download = 'document.docx'; // 强制下载而非打开

      document.body.appendChild(link);

      link.click();

      document.body.removeChild(link);

    }

    window.onload = downloadFile;

  </script>

  

  <p>如果未自动下载，<a href="document.docx" download>点击此处</a></p>
